> I know that you check for it in 'AH', but don't know which number it is?

See source/features/screen.asm in MikeOS; specifically the
os_list_dialog routine:

.another_key:
        call os_wait_for_key            ; Move / select option
        cmp ah, 48h                     ; Up pressed?
        je .go_up
        cmp ah, 50h                     ; Down pressed?
        je .go_down
        cmp al, 13                      ; Enter pressed?
        je .option_selected
        cmp al, 27                      ; Esc pressed?
        je .esc_pressed
        jmp .more_select                ; If not, wait for another key
