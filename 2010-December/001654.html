Just puttin' this out there for anybody else who might be new to OS
dev in assembly:

mov   ax, filename
mov   bx, 0
mov   cx, 32768
call    os_load_file

mov   ax, 0    ; argument passed in reg ax
mov   bx, 0    ; arg in bx
mov   cx, 0    ; cx
mov   dx, 0   ; dx
mov   si, 0    ; si
mov   di, 0    ; di

call    32768   ; start program

filename   db 'TESTAPP.BIN',0
