Hi all...

I now have full colour printing using existing MikeOS...

The BASIC code is below...
(And the assembly code below that.)
====================================================

REM A simple means of colour text/characters for MikeOS 4.1x.
REM Idea (C)2010, B.Walker, G0LCU, issued as PD for MikeOS.

REM Enter the machine code subroutine for colour control...
POKE 156 65500
POKE 80 65501
POKE 83 65502
POKE 81 65503
POKE 184 65504
POKE 32 65505
POKE 9 65506
POKE 183 65507
POKE 0 65508
POKE 179 65509
POKE 7 65510
POKE 185 65511
POKE 1 65512
POKE 0 65513
POKE 205 65514
POKE 16 65515
POKE 89 65516
POKE 91 65517
POKE 88 65518
POKE 157 65519
POKE 195 65520

REM Printing a character string in multi-colours.
FOR n = 0 TO 79
REM Position to print a simple series of characters.
MOVE n 7

REM My character to be printed.
POKE 221 65505

REM Colour to print, 16 * b for background, b = 0 to 7, (15).
REM Colour to print, f for foreground f = 0 to 15.
REM Red background and bright white foreground.
POKE 79 65510

REM Call the simple print routine...
CALL 65500
NEXT n

REM Exit as normal...
END

====================================================

;Simple colour control for MikeOS 4.0x and above.
;start:
	pushf			;Save used registers.
	push	ax		;
	push	bx		;
	push	cx		;
	mov	ax,0920h		;Set the character to SPACE for interrupt
				;using function ah = 09h...
	mov	bh,00h		;Set the page to be written to.
	mov	bl,07h		;Set to white on black.
	mov	cx,01h		;Just prine one character unless changed.
	int	10h		;Trigger the interrupt.
	pop	cx		;Return registers back
	pop	bx		;
	pop	ax		;
	popf			;
	ret			;Return to the calling routine...
;end:

====================================================
