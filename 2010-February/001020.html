> Real time KB access for the two caps shift keys for input.
> 
> There are NO interrupts needed for the assembly code...
> 
> AD86 assembly code first...
> 
> 	pushf
> 	push	ax
> 	push	dx
> 	push	ds
> 	mov	dx,0040h
> 	mov	ds,dx
> 	mov	al,[0017h]
> 	pop	ds
> 	pop	dx
> 	mov	[store],al
> 	pop	ax
> 	popf
> 	ret
> store	db	0c3h
>
