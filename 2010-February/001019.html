Hi all...

Real time KB access for the two caps shift keys for input.

There are NO interrupts needed for the assembly code...

AD86 assembly code first...

	pushf
	push	ax
	push	dx
	push	ds
	mov	dx,0040h
	mov	ds,dx
	mov	al,[0017h]
	pop	ds
	pop	dx
	mov	[store],al
	pop	ax
	popf
	ret
store	db	0c3h

Now the Mike BASIC code...

POKE 156 65500
POKE 80 65501
POKE 82 65502
POKE 30 65503
POKE 186 65504
POKE 64 65505
POKE 0 65506
POKE 142 65507
POKE 218 65508
POKE 160 65509
POKE 23 65510
POKE 0 65511
POKE 31 65512
POKE 90 65513
POKE 162 65514
POKE 240 65515
POKE 255 65516
POKE 88 65517
POKE 157 65518
POKE 195 65519
POKE 195 65520
begin:
CALL 65500
PEEK c 65520
PRINT c ;
PRINT " " ;
IF c = 3 THEN GOTO cleanexit
GOTO begin
cleanexit:
PAUSE 20
END
