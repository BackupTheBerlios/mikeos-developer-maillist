Hi Mike,

Herewith a possible addition to both MikeOS and MikeOS BASIC

For BASIC, I was thinking of a new command like such:

PRINTCOLOUR text_colour, "actual_text"

Which would call os_print_colour_string and keep compatibility with the existing
PRINT command in BASIC and os_print_string.

; -----------------------------------------------------------------------
; os_print_colour_string -- displays text with colour (range = 0 to 15)
;
; IN: BL = text colour; SI = message location (zero-terminated string)
; -----------------------------------------------------------------------

os_print_colour_string:

    pusha

    mov ah, 09h            ; int 10h write character and attribute function
    mov bh, 0            ; page number -- normally zero
    mov cx, 1            ; number of times to print each character

.repeat:
    lodsb                ; Get char from string
    cmp al, 0
    je .done            ; If char is zero, end of string

    int 10h                ; Otherwise, print it
    jmp .repeat            ; And move on to next char

.done:
    popa
    ret

; -------------------------------------------------------------------------
